sockets.Client {
    /login (POST): {
        nickname: string
    },

    /logout (GET): {
        nickname: string
    }

    /list (GET): {
        nickname: string
    }

    /upload (POST): {
        nickname: string,
        whom: string,
        file: string
    }

    /send (GET) {
        nickname: string,
        client: DESKTOP|MOBILE
    }

    {
        action: "handshake",
        data: {
            nickname: string,
            client: DESKTOP|MOBILE
        }
    }

    {
        action: "message",
        data: {
             whom: string,
             message: string
        }
    }
}
